362 void Polyhedron::create_edge(Vertex* v1, Vertex* v2)
363 {
364  int i, j;
365  Triangle* f;
366 
367  /* make sure there is enough room for a new edge */
368 
369  if (nedges >= max_edges) {
370 
371  max_edges += 100;
372  Edge** list = new Edge * [max_edges];
373 
374  /* copy the old list to the new one */
375  for (i = 0; i < nedges; i++)
376  list[i] = elist[i];
377 
378  /* replace list */
379  free(elist);
380  elist = list;
381  }
382 
383  /* create the edge */
384 
385  elist[nedges] = new Edge;
386  Edge* e = elist[nedges];
387  e->index = nedges;
388  e->verts[0] = v1;
389  e->verts[1] = v2;
390  nedges++;
391 
392  /* count all triangles that will share the edge, and do this */
393  /* by looking through all faces of the first vertex */
394 
395  e->ntris = 0;
396 
397  for (i = 0; i < v1->ntris; i++) {
398  f = v1->tris[i];
399  /* examine the vertices of the face for a match with the second vertex */
400  for (j = 0; j < 3; j++) {
401  /* look for a match */
402  if (f->verts[j] == v2) {
403  e->ntris++;
404  break;
405  }
406  }
407  }
408 
409  /* make room for the face pointers (at least two) */
410  if (e->ntris < 2)
411  e->tris = new Triangle * [2];
412  else
413  e->tris = new Triangle * [e->ntris];
414 
415  /* create pointers from edges to faces and vice-versa */
416 
417  e->ntris = 0; /* start this out at zero again for creating ptrs to tris */
418 
419  for (i = 0; i < v1->ntris; i++) {
420 
421  f = v1->tris[i];
422 
423  /* examine the vertices of the face for a match with the second vertex */
424  for (j = 0; j < 3; j++)
425  if (f->verts[j] == v2) {
426 
427  e->tris[e->ntris] = f;
428  e->ntris++;
429 
430  if (f->verts[(j + 1) % 3] == v1)
431  f->edges[j] = e;
432  else if (f->verts[(j + 2) % 3] == v1)
433  f->edges[(j + 2) % 3] = e;
434  else {
435  fprintf(stderr, "Non-recoverable inconsistancy in create_edge()\n");
436  exit(-1);
437  }
438 
439  break; /* we'll only find one instance of v2 */
440  }
441 
442  }
443 }

566 /******************************************************************************
567 Order the pointers to faces that are around a given vertex.
568 
569 Entry:
570  v - vertex whose face list is to be ordered
571 ******************************************************************************/
572 
573 void Polyhedron::order_vertex_to_tri_ptrs(Vertex* v)
574 {
575  int i, j;
576  Triangle* f;
577  Triangle* fnext;
578  int nf;
579  int vindex;
580  int boundary;
581  int count;
582 
583  nf = v->ntris;
584  f = v->tris[0];
585 
586  /* go backwards (clockwise) around faces that surround a vertex */
587  /* to find out if we reach a boundary */
588 
589  boundary = 0;
590 
591  for (i = 1; i <= nf; i++) {
592 
593  /* find reference to v in f */
594  vindex = -1;
595  for (j = 0; j < f->nverts; j++)
596  if (f->verts[j] == v) {
597  vindex = j;
598  break;
599  }
600 
601  /* error check */
602  if (vindex == -1) {
603  fprintf(stderr, "can't find vertex #1\n");
604  exit(-1);
605  }
606 
607  /* corresponding face is the previous one around v */
608  fnext = other_triangle(f->edges[vindex], f);
609 
610  /* see if we've reached a boundary, and if so then place the */
611  /* current face in the first position of the vertice's face list */
612 
613  if (fnext == NULL) {
614  /* find reference to f in v */
615  for (j = 0; j < v->ntris; j++)
616  if (v->tris[j] == f) {
617  v->tris[j] = v->tris[0];
618  v->tris[0] = f;
619  break;
620  }
621  boundary = 1;
622  break;
623  }
624 
625  f = fnext;
626  }
627 
628  /* now walk around the faces in the forward direction and place */
629  /* them in order */
630 
631  f = v->tris[0];
632  count = 0;
633 
634  for (i = 1; i < nf; i++) {
635 
636  /* find reference to vertex in f */
637  vindex = -1;
638  for (j = 0; j < f->nverts; j++)
639  if (f->verts[(j + 1) % f->nverts] == v) {
640  vindex = j;
641  break;
642  }
643 
644  /* error check */
645  if (vindex == -1) {
646  fprintf(stderr, "can't find vertex #2\n");
647  exit(-1);
648  }
649 
650  /* corresponding face is next one around v */
651  fnext = other_triangle(f->edges[vindex], f);
652 
653  /* break out of loop if we've reached a boundary */
654  count = i;
655  if (fnext == NULL) {
656  break;
657  }
658 
659  /* swap the next face into its proper place in the face list */
660  for (j = 0; j < v->ntris; j++)
661  if (v->tris[j] == fnext) {
662  v->tris[j] = v->tris[i];
663  v->tris[i] = fnext;
664  break;
665  }
666 
667  f = fnext;
668  }
669 }
